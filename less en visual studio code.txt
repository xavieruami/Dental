para compilar automáticamente
instalar Gulp escribiendo el el cmd la siguiente instrucción
npm install gulp gulp-sass gulp-less
presionar Enter
En la raiz del directorio donde está el archivo less crear el archivo gulpfile.js
// Less configuration
var gulp = require('gulp');
var less = require('gulp-less');

gulp.task('less', function() {
    gulp.src('*.less')
        .pipe(less())
        .pipe(gulp.dest(function(f) {
            return f.base;
        }))
});

gulp.task('default', ['less'], function() {
    gulp.watch('*.less', ['less']);
})
guardar

el archivo task.json se debe editar para que quede como
{
    "version": "0.1.0",
    "command": "gulp",
    "isShellCommand": true,
    "tasks": [
        {
            "taskName": "default",
            "isBuildCommand": true,
            "showOutput": "always",
            "isWatching": true
        }
    ]
}
